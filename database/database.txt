1- CREATE DATABASE comunidade_anima;

2- CREATE TABLE usuario(id int(9) auto_increment, primary key(id), ra int(9) unique key, email varchar(50), nome varchar(70), imagem_perfil varchar(100));

3- CREATE TABLE administrador(id int(9) auto_increment, primary key(id), usuario varchar(30), senha varchar(100) , nome varchar(70), email varchar(50), nivel varchar(4));

4- CREATE TABLE eventos(id INT (9) AUTO_INCREMENT PRIMARY KEY, titulo VARCHAR(50), data DATE, horario TIME, endereco VARCHAR(100), descricao TEXT, arquivos VARCHAR(400), situacao_post VARCHAR(10), justificativa TEXT, autor INT, FOREIGN KEY (autor) REFERENCES usuario(ra));

5- CREATE TABLE curtidas(curtido BOOLEAN, id_posts int(9) foreign key references eventos(id), id_ususario foreign key references usuario(id));

5.1(Corrigida) - CREATE TABLE curtidas(curtido BOOLEAN, id_posts INT(9), FOREIGN KEY (id_posts) REFERENCES eventos(id), id_usuario INT, FOREIGN KEY (id_usuario) REFERENCES usuario(id));

6- CREATE TABLE servicos_universitarios(id int(6) auto_increment primary key, servico varchar(30), descricao text(500), telefone varchar(100), email varchar(50))
